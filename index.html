var, let, and const

Let : It has Block scope , can be over written
Const : It has Block scope , can not be over written

Array Destructuring - use of squary bracket

its a js expression that makes it possible to unpack values from arrays, or properties from array, into distinct variables 


const languages = ['js','html','css','php'];
// traditional way of accessing js array elements
// let top1 = languages[0]; let top2 = languages[1]; let top3 = languages[2]; let top4 = languages[3];

// ES6 - Sequence does matter
let [ top1, top2, top3, top4 ] = languages;
console.log( `elements are : ${top1}, ${top2}, ${top3}, ${top4}` );

// We can skip elements also
let [ fav, , , leastFav ] = languages;
console.log( `favourite : ${fav} AND Least Favourite : ${leastFav}` );

// get first two elements
let [top5,top6] = languages;
console.log(`top5 : ${top5} AND top6 : ${top6}`);
